# [电影节](http://bailian.openjudge.cn/practice/4151/)

```C++
#include <iostream>
#include <algorithm>
using namespace std;

struct film {
    int begin;
    int end;
} f_list[100];

int cmp(film a, film b) {
    return a.end < b.end;
}

int main() {
    int n;
    while (cin >> n && n != 0) {
        for (int i = 0; i < n; i++) {
            cin >> f_list[i].begin >> f_list[i].end;
        }
        sort(f_list, f_list + n, cmp);
        int sum = 1;//若定义在while之前则WA
        int end_time = f_list[0].end;
        for (film f : f_list) {
            if (f.begin >= end_time) {
                end_time = f.end;
                sum++;
            }
        }
        cout << sum << endl;
    }
    return 0;
}
```
