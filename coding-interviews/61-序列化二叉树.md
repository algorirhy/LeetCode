# [序列化二叉树](https://www.nowcoder.com/practice/cf7e25aa97c04cc1a68c8f040e71fb84?tpId=13&tqId=11214&tPage=4&rp=4&ru=/ta/coding-interviews&qru=/ta/coding-interviews/question-ranking)

```c++
class Solution {
public:
    char* Serialize(TreeNode* root) {
        if(!root)
            return nullptr;
        string str;
        queue<TreeNode*> q;
        q.push(root);
        while(!q.empty()){
            int n = q.size();
            for(int i = 0; i < n; i++){
                if(q.front()){
                    q.push(q.front()->left);
                    q.push(q.front()->right);
                    str += to_string(q.front()->val) + ' ';
                }else{
                    str += '# ';
                }
                q.pop();
            }
        }
        char* ch = strdup(str.c_str());
        return ch;
    }
    
    TreeNode* Deserialize(char* str) {
        if(!str)
            return nullptr;
        int k = 0;
        TreeNode* root = nextNode(str, k);
        queue<TreeNode*> q;
        q.push(root);
        while(!q.empty()){
            int n = q.size();
            for(int i = 0; i < n; i++){
                q.front()->left = nextNode(str, k);
                q.front()->right = nextNode(str, k);
                if(q.front()->left)
                    q.push(q.front()->left);
                if(q.front()->right)
                    q.push(q.front()->right);
                q.pop();
            }
        }
        return root;
    }
    
    TreeNode* nextNode(char* str, int& i){
        string s;
        //多位数
        while(str[i] != ' '){
            if(str[i] == '#'){
                i += 2;
                return nullptr;
            }
            s += str[i];
            i++;
        }
        i++;
        if(!s.empty())
            return new TreeNode(stoi(s));
        return nullptr;
    }
};
```
