# 方法一

递归

```c++
class Solution {
public:
    void Mirror(TreeNode *pRoot) {
        if(pRoot == nullptr)
            return;
        TreeNode* temp = pRoot->left;
        pRoot->left = pRoot->right;
        pRoot->right = temp;
        if(pRoot->left)
            Mirror(pRoot->left);
        if(pRoot->right)
            Mirror(pRoot->right);
    }
};
```



# 方法二

非递归

```c++
class Solution {
public:
    void Mirror(TreeNode *pRoot) {
        if(pRoot == nullptr)
            return;
        stack<TreeNode*> stk;
        stk.push(pRoot);
        while(!stk.empty()){
            TreeNode* node = stk.top();
            stk.pop();
            if(node->left || node->right){
                TreeNode* temp = node->left;
                node->left = node->right;
                node->right = temp;
            }
            if(node->left)
                stk.push(node->left);
            if(node->right)
                stk.push(node->right);
        }
    }
};
```
