动态规划

```c++
class Solution {
public:
    int FindGreatestSumOfSubArray(vector<int> array) {
        if(!array.size())
            return 0;
        vector<int> dp;
        dp.push_back(array[0]);
        for(int i = 1; i < array.size(); i++){
            dp.push_back(max(array[i], array[i] + dp[i - 1]));
        }
        int res = 0;
        for(int i = 1; i < array.size(); i++){
            if(dp[i] > dp[res])
                res = i;
        }
        return dp[res];
    }
};
```



```c++
class Solution {
public:
    int FindGreatestSumOfSubArray(vector<int> array) {
        int res = array[0];
        int max_ = array[0];
        for(int i = 1; i < array.size(); i++){
            max_ = max(array[i], max_ + array[i]);
            res = max(max_, res);
        }
        return res;
    }
};
```
